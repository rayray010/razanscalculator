  <script>
    let pyodide;
    async function loadPy() {
      pyodide = await loadPyodide();
      document.getElementById("result").textContent = "الآلة جاهزة!";
    }
    loadPy();

    document.getElementById("calcBtn").onclick = async () => {
      const n1 = document.getElementById("num1").value;
      const n2 = document.getElementById("num2").value;
      const op = document.getElementById("op").value;
      const code = `
def calc(a, op, b):
    try:
        a = float(a)
        b = float(b)
    except:
        return "خطأ: أدخل أعدادًا صحيحة"
    if op == "+": return a + b
    elif op == "-": return a - b
    elif op == "*": return a * b
    elif op == "/":
        return "اقتسام العدد على عدمٍ محال" if b == 0 else a / b
    elif op == "**": return a ** b
    else: return "غير متاح"
calc(${JSON.stringify(n1)}, ${JSON.stringify(op)}, ${JSON.stringify(n2)})
`;
      const result = await pyodide.runPythonAsync(code);
      document.getElementById("result").textContent = "النتيجة: " + result;
    };
  </script>
